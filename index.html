Get Started

Quick installation and first prediction in minutes

# Install ChanFM
pip install -U pip
pip install chanfm

# Or install from source
git clone https://github.com/channelfm/chanfm.git
cd chanfm
pip install -e .
# Quick prediction
from chanfm.inference import api

sample = {
    "env": "UMi",
    "frequency_Hz": 28e9,
    "tx": {"pos_m": [0, 0, 10], "power_dBm": 23},
    "rx": {"pos_m": [50, 0, 1.5]},
    "ant": {"tx_M": 8, "rx_N": 8}
}

result = api.predict(sample, task=["path_loss_dB", "H"])
print(f"Path loss: {result['path_loss_dB']:.2f} dB")
# Complete workflow example
import chanfm
from chanfm.data import load_dataset
from chanfm.models import ChanFM

# Load pre-trained model
model = ChanFM.from_pretrained("chanfm-base")

# Load your data
data = load_dataset("your_dataset.json")

# Fine-tune for your scenario
model.finetune(data, epochs=10)

# Export for deployment
model.export("chanfm_custom.onnx")
